\def\PropertiesA{
	Important properties:
	\begin{itemize}
		\item Order, $n = |V(G)| \geq 1$
		\item Size, $m = |E(G)|\in [0, (n-1)n/2]$
		\item Neighbourhood, $N(v)$ is the set of vertices adjacent to $v$
		\item Degree, $\deg(v) = |N(v)|$
		$$0 \leq \delta (G) \leq \deg(v) \leq \Delta (G) \leq n-1$$
	\end{itemize}
	\phantom{}
}

\def\PropertiesB{
	Important families of graphs:
	\begin{itemize}
		\item $N_n$ is a null graph of order $n$ and size 0
		\item $P_n$ is a path s.t. $V=\{v_1,v_2,\dots,v_n\}$ and $E=\{v_1v_2, v_2v_3, \dots, v_{n-1}v_n\}$
		\item $C_n$ is a cycle s.t.  $V=\{v_1,v_2,\dots,v_n\}$ and $E=\{v_1v_2, v_2v_3, \dots, v_{n-1}v_n, v_nv_1\}$
		\item $K_n$ is a complete graph of order $n$ and maximum size
		\item $\bar{G}$ of $G$ is the compliment of $G$ s.t. $E(\bar{G}) = E(K_n) - E(G)$
		\item $r$-regular s.t. each vertex has degree $r$ and size $rn/2$
		\item $K_{x,y}$ is a complete bipartite graph s.t. $V = V_x \cup V_y$ s.t. every vertex of the $V_x$ is connected to every vertex of $V_y$.
	\end{itemize}
}

\def\PropertiesC{
	Some important operations of graphs are
	\begin{itemize}
		\item $H$ is a subgraph of $G$ if $V(H) \subseteq V(G)$ and/or $E(H) \subseteq E(G)$. 
		\item $G-e$ is a spanning graph if $V(G-e) = V(G)$ and $E(G-e) \subseteq E(G)$.
		\item $G-v$ is an induced graph if $V(G-v) \subseteq V(G)$ and $E(G-v) = E(G) - (V(G) - V(G-v))$.
		\item $G+H = F$ s.t. $V(F) = V(G) \cup V(H)$ and $E(F) = E(G) \cup E(H) \cup \{gh: g \in E(G), h \in E(H)\}$.
		\item $G \times H = F$ s.t. $V(F) = V(G) \times V(H)$ and $E(F) = \{\{gh,g'h'\}: g = g' \land N(h)=h' \text{ or vice versa}\}$.
	\end{itemize}
}

\def\Sequence{
	A $u-v$ walk is any sequence of vertices, i.e. $W:u=u_0, u_1, u_2, \dots, u_k = v$. \\
	\phantom{} \\
	A $u-v$ trail is a walk in which no edge is repeated, i.e. $T:u=u_0, u_1, u_2, \dots, u_k = v$ where $u_{i-1}u_{i}\not = u_{j-1}u_j$ where $1 \leq i,j \leq k$. A closed trail of length 3 or more is known as a circuit. \\
	\phantom{} \\
	A $u-v$ path is a trail in which no vertex is repeated, i.e.  $P:u=u_0, u_1, u_2, \dots, u_k = v$ where $u_i\not = u_j$ where $0 \leq i,j \leq k$. A closed path of length 3 or more is known as a cycle. The shortest path between two vertices is known as a geodesic with distance $d(u,v)$. And the maximum distance in a graph is known as the diameter of the graph, $\text{diam}(G)$.
}

\def\Connected{
	The first theorem of Graph theory (a.k.a. Handshake lemma) states that $\underset{v \in V}{\sum}\deg(v) = 2m$. \\
	\phantom{} \\
	If there exists a path between every pair of vertices in a graph, then it is a connected graph. \\
	\phantom{} \\
	If a graph is disconnected, then it is the union of its components, i.e. $G=G_1 \cup G_2 \cup \dots \cup G_n$ with $n$ number of components, i.e. $k(G)=n$. \\
	\phantom{} \\
	Sufficient condition: if $\deg(u) + \deg(v) \geq n-1$ $\forall u,v \in V(G)$ s.t. $uv \not\in E(G)$, then $G$ is connected and has $\text{diam}(G) \leq 2$. \\
	\phantom{} \\
	$G$ is isomorphic to $H$, ($G \cong H$), if there exists a bijection $\varphi: V(G) \leftrightarrow V(H)$ s.t. if $uv \in E(G)$ then $\varphi(u)\varphi(v) \in E(H)$.
}
	
\def\Degree{
	Degree Sequence, $s$ is the sequence of non-increasing degrees of $G$. A sequence is graphical if it is a degree sequence \\
	\phantom{} \\
	Havel-Hakimi theorem: $s:d_1, d_2, d_3, \dots d_n$ a.t. $n \geq 2$ and $d \geq 0$ is graphical if $d_2-1, d_3-1, \dots, d_{d_1+1}-1, d_{d_1+2}, \dots d_n$ \\
	\phantom{} \\
	Isomorphic Graphs, $G \cong H$ if they have the same structure ($n,m,deg, N, k-cycles$), i.e. there exists a bijection $\phi: V(G) \rightarrow V(H)$ s.t. if $uv \in E(G)$ then $\phi(u)\phi(v) \in E(H)$. A graph is self complimentary if $G \cong \bar{G}$
}

\def\Tree{
	A \textbf{bridge} is an edge which if removed causes the graph to become disconnected. \\
	\phantom{} \\
	\textbf{theorem:} An edge $e \in E(G)$ is a bridge if and only if $e$ lies on no cycle of $G$. \\
	\phantom{} \\
	A \textbf{tree}, $T$, is an acyclic connected graph in which every edge is a bridge. \\
	\phantom{} \\
	\textbf{theorem:} A graph $G$ is a tree if and only if every two vertices of $G$ are connected by a unique path. \\
	\phantom{} \\
	\textbf{theorem:} Every nontrivial tree has at least two end vertices.  \\
	\phantom{} \\
	\textbf{theorem:} Every tree of order $n$ has size $n-1$. \\
	\phantom{} \\
	Another class of acyclic graphs are \textbf{forests} in which each component is a tree, i.e. $F = T_1 \cup T_2 \cup \dots \cup T_n$. \\
	\phantom{} \\
	Similar to bridges, a \textbf{cut vertex} is a vertex which if removed causes the tree to become disconnected, i.e. $v \in V(G)$ is a cut vertex if $G$ is connected and $G-v$ is disconnected.  \\
	\phantom{} \\
	\textbf{theorem:} Let $v$ be a vertex incident with a bridge in a connected graph $G$. Then $v$ is a cut vertex if and only if $\deg(v) \geq 2$. \\
}

\def\MST{
	If $G$ is connected and $H$ is spanning subgraph of $G$ and a tree, then $H$ is a \textbf{spanning tree} of $G$. \\
	\phantom{} \\
	\textbf{theorem:} Every connected graph contains a spanning tree.\\
	\phantom{} \\
	If each edge in a graph $G$ is assigned a weight, $w(e)$, then the graph is a \textbf{weighted graph} and the weight of the graph is $$w(G) = \underset{e \in E(G)}{\sum} w(e)$$ \\
	\phantom{} \\
	The minimum spanning tree $H$ of a weighted connected $G$ is the spanning tree with the minimum $w(H)$ among all spanning trees of $G$. Two of the most prominent methods for determining the minimum spanning tree are Kruskal's algorithm and Prism's algorithm. \\
	\phantom{} \\
	\textbf{Kruskal's algorithm:} Suppose $G$ is a connected graph such that $E(G) = \{e_1, e_2, \dots, e_n\}$ where $w(e_1) \leq w(e_2) \leq \dots \leq w(e_n)$. Then its minimum spanning tree can be constructed using the following algorithm. \\
	\begin{enumerate}
		\item Make a null graph $H$ such that $V(H) = V(G)$.
		\item Add $e_1$ to $H$, i.e. $E(H) = E(H) \cup e_1$.
		\item Continue adding $e_i$ from $G$ to $H$ in increasing order - ensuring no cycles are formed - till all vertices are connected.
		\item If two edges have the same weight then choose the most suitable one. 
	\end{enumerate}
	\phantom{} \\
	\textbf{Prism's algorithm:} Suppose $G$ is a connected graph such that $E(G) = \{e_1, e_2, \dots, e_n\}$ where $w(e_1) \leq w(e_2) \leq \dots \leq w(e_n)$. Then its minimum spanning tree can be constructed using the following algorithm. 	
	\begin{enumerate}
		\item Make a null graph $H$ such that $V(H) = V(G)$.
		\item Choose an arbitrary vertex $v$.
		\item Check the weights of all edges adjacent to $v$ and choose the edge with minimum weight ensuring that no cycle is formed.
		\item Continue till all vertices are connected.
		\item If two edges have the same weight then choose the most suitable one.
	\end{enumerate}
}

\def\Eulerian{
	A circuit $C$ in a connected graph $G$ is an \textbf{Eulerian circuit} if it contains every edge of $G$, e.g. $C_n$. If a graph contains an Eulerian circuit, it is a \textbf{Eulerian graph}. And since a circuit is a closed trail, no edge is repeated. If the trail is open, then it is known as an \textbf{Eulerian trail}, e.g. $P_n$. A graph with only a Eulerian trial is known as a \textbf{Semi-Eulerian graph}. \\
	\phantom{} \\
	\textbf{theorem:} A nontrivial connected graph $G$ is Eulerian if and only if every vertex of $G$ has even degree. \\
	\phantom{} \\
	\textbf{corollary:} A connected graph $G$ contains an Eulerian trial if and only if exactly two vertices of $G$ have odd degree. Furthermore, each Eulerian trail of $G$ begins at one of these odd vertices and ends at the other.  \\
}

\def\Hamiltonian{
	Similar to Eulerian graphs, a cycle $C$ in a graph $G$ which contains every vertex of $G$ is a \textbf{Hamiltonian cycle}, e.g. $C_n$ and $K_n$. And a graph containing a Hamiltonian cycle is a \textbf{Hamiltonian graph}. And since a cycle is a closed path, no vertex is repeated. If the path is open then that it is a \textbf{Hamiltonian path}, e,g, $P_n$. A graph containing a Hamiltonian path is known as \textbf{Semi-Hamiltonian graph}. \\
	\phantom{} \\
	Every Hamiltonian graph contains a Hamiltonian cycle $C$ which is 2-regular. And every proper subgraph of $C$ is a path or a union of paths meaning $C$ does not contain any other cycle or a subgraph with vertex of degree $\geq$ 2. Moreover, since $C$ is a 2-regular subgraph of $G$, if $G$ contains a vertex of degree 2, then both edges incident on that vertex lie in $C$. \\
	\phantom{} \\
	\textbf{theorem:} The Petersen graph is non-Hamiltonian. \\
	\phantom{} \\
	\textbf{theorem:} If $G$ is a Hamiltonian graph, then for every nonempty proper set $S$ of vertices  of $G$, $k(G-S) \leq |S|$ \\
	\textbf{corollary}: If $G$ contains at least one cut vertex, it is non Hamiltonian \\
	\phantom{} \\
	Sufficient Conditions: \\
	\textbf{Ore Condition:} Let $G$ be a graph of order $n \geq 3$. If $\deg(u) + \deg(v) \geq n$ for each pair of non adjacent vertices of $G$, then $G$ is Hamiltonian. \\
	\textbf{Direc Condition:} Given a graph $G$ of order $n \geq 3$, if $\deg(v) \geq n/2$ $\forall v \in V(G)$, then $G$ is Hamiltonian. 
}

\def\Planar{
	A graph $G$ is \textbf{planar} if it can be drawn on a plane without intersecting edges. Hence the graph with no intersecting edges is known as a \textbf{plane graph}. Examples of planar graphs include $C_n$, $P_n$, $K_{1,n}$ and trees. A graph $G$ is \textbf{maximal planar} if $G$ is planar but the addition of an edge between any two non adjacent vertices of $G$ result in a non-planar graph. For $1 \leq n \leq 4$, the only maximal planar graphs are $K_n$. \\
	\phantom{} \\
	A plane graph divides the plane into \textbf{regions}. The vertices which connects regions are known as \textbf{boundaries}. Moreover, the region which contains all external vertices of $G$ is known as the \textbf{exterior region}.
	\phantom{} \\
	If a plane graph is connected, then some useful observations can be made. If it contains a bridge, then that bridge appears in only one region. All other edges lie on the boundary of two regions. If an edge is removed, then the graph remains plane but has one less region. Moreover for plane graphs of $m \geq 3$, each region has at least 3 edges. \\
	\phantom{} \\
	\textbf{theorem:} \textbf{(Euler Identity)}: If $G$ is a connected plane graph of order $n$, size $m$ and having $r$ regions, then $n-m+r=2$ \\
	\textbf{theorem:} If $G$ is a planar graph of order $n \geq 3$ and size $m$, then $m \leq 3n-6$ \\
	\textbf{corollary:} Every planar graph contains a vertex of degree 5 or less. \\
	\textbf{corollary:} $K_5$ is non-planar. \\
	\textbf{theorem:} The graph $K_{3,3}$ is non-planar.  \\
	\phantom{} \\
	A subdivision $G'$ of a graph $G$ is formed when one or more vertices of degree 2 are inserted into one or more edges of $G$. In the example below, $G_1, G_2$ and $G_3$ are subdivisions of $G$.
	\phantom{} \\
	\textbf{theorem:} \textbf{(Kuratowski's Theorem)}: A graph $G$ is planar if and only if´ $G$ does not contain $K_5$, $K_{3,3}$ or a subdivision of them as a subgraph.
}

\def\Color{
	Each map has an associated graph $G$, known as the \textbf{dual} of the map whose vertices are the regions and edges are neighbouring regions. $G$ is always a connected planar graph and the converse holds true as well, that is every connected planar graph is the dual of some map. \\
	\phantom{} \\
	In a \textbf{proper colouring} of $G$, each vertex is assigned a colour s.t. all adjacent vertices have different colours. The smallest number of colours used to do so is known as the \textbf{chromatic number} of $G$, $\chi(G)$. Suppose $k$ colours can be used for colouring $G$, then $G$ is said to be \textbf{$k$-colourable} and a colouring using $k$ colours is called a \textbf{$k$-colouring} and $G$ is \textbf{$k$-chromatic}. Suppose $\chi(G)=k$, then every $k$-colouring of $G$ is a \textbf{minimum colouring} of $G$.  \\
	\phantom{} \\
	\textbf{theorem:} A graph $G$ has $\chi(G) = 2$ if and only if $G$ is a nonempty bipartite graph. \\
	\textbf{theorem:} For every graph $G$, $\chi(G) \leq 1 + \Delta(G)$ \\
	\phantom{} \\
	\phantom{} \\
	Similar to vertex colouring, in an \textbf{edge colouring} of a nonempty graph $G$, each edge is assigned a colour s.t. all adjacent edges are of different colours. The minimum number of colours which can be used is known as the \textbf{chromatic index} or \textbf{chromatic number} and is denoted by $\chi_1(G)$ and an edge colouring with $k$ is known as a \textbf{$k$-edge colouring} \\
	\phantom{} \\
	\textbf{theorem:} \textbf{Vizing's Theorem:} $\chi_1(G) = \Delta(G)$ or $\chi_1(G) = 1 + \Delta(G)$ \\
}

\def\Digraph{
 	A \textbf{directed graph} or \textbf{digraph} is a graph $D$ with a finite nonempty vertex set $V$ and a set $E$ containing ordered pairs of $V$ known as \textbf{arcs} or \textbf{directed edges}. \\
 	\phantom{} \\
	If a directed graph has the property that for a pair of vertices $u,v \in V$, only one of $(u,v), (v,u) \in E$´, then it is an \textbf{oriented graph}. Alternatively, an oriented graph can be obtained by assigning directions to edges of a graph $G$; then the digraph is an \textbf{orientation} of $G$. If both $(v,u) \in E$ for all $(u,v) \in E$, then the digraph is \textbf{symmetric}. \\	
	\phantom{} \\
	If $(u,v) \in E$, then both $u,v$ are incident with each other with $u$ being adjacent to $v$ and $v$ being adjacent from $u$. The number of vertices to which $v$ is adjacent is the \textbf{outdegree} of $v$, $\text{od}(v)$ whereas the number of vertices from which $v$ is adjacent is the \textbf{indegree} of $v$, $\text{id}(v)$. Hence the first theorem of Digraph theory states $$\underset{v \in V}{\sum} \text{od}(v) = \underset{v \in V}{\sum} \text{in}(v) = m$$ \\
	\phantom{} \\
	A \textbf{tournament} is an orientation of a complete graph, that is, for all pairs $u,v$ of distinct vertices only one of $(u,v), (v,u)$ is an arc.  If $(u,v), (v,w) \in E$, then the tournament is transitive, i.e. $(u,w) \in E$. \\
	\phantom{} \\
	\textbf{theorem:} Every tournament contains a Hamiltonian path. \\
}
